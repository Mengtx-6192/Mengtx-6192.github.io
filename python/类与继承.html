# 类与继承

# 一.类

class Car():
	"""docstring for Car"""
	wheel = 4
	def __init__(self, arg):
		print('init')
		self.arg = arg
	def drive(self):
		print('开车', self.arg)
BMW = Car(1)
print(BMW)          # init
print(BMW.wheel)    # 4
print(BMW.drive())  # 开车

# 二.继承

# 1.多层继承

class Car():
	wheel = 4

class BMW(Car):
	def drive(self):
		print('开车')

class BMW3(BMW):
	pass

BMW111 = BMW3()
print(BMW111.wheel)    # 4
print(BMW111.drive())  # 开车

# 2.多重继承

class Car1():
	wheel = 4

class Car2():
	wheel = 8

class Car(Car1, Car2):
	pass

BMW = Car()
print(BMW.wheel) # 8

# 三.拓展

# 上面的BMW案例，首先是创建了个实例BMW111， 然后再调用drive方法。

class Car():
	wheel = 4

class BMW(Car):
	def __init__(self, arg):
		self.arg = arg
	def __str__(self):
		return self.arg

class BMW3(BMW):
	pass

BMW111 = BMW3('开车')
print(BMW111) # 开车

# __str__类中的特殊方法，可以在创建实例后自执行，注意和__init__的区别
