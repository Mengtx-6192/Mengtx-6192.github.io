# require 和 import

# require

// js
const list = [1,2,3]
let a = false

// 1
moudle.exports = {
    ...list,
    a
}
// 2
exports.list1 = list
exports.a = a

// 文件中引入
const res = require(./xxx.js)
// 1.
console.log(res) // { list:[1,2,3], a: false }
// 2
console.log(res) // { list1:[1,2,3], a: false }

# import

const list = [1,2,3]
let a = false

export default = {
    list,
    a
}
export { list }

// 文件中引入
import res, { a } from './xxx.js'
console.log(res, a) // { list: [1,2,3], a: false }, false
